void Dec_to_Bin(){
    int number;
    long long res = 0;
  int reminder, j = 1;
    printf("Enter a Decimal Number : \n");
    scanf("%d",&number);
  while (number!=0) {
    reminder = number % 2;
    number = number/ 2;
    res += reminder * j;
    j *= 10;
  }
  printf("Binary number = %lld",res);
}

void Dec_to_Oct(){

    scanf("%d",&input);

        output += (input % 8) * j;
        input = input/8;
        j *= 10;
    }

    printf("Octal value =  %d",output);
}
void Dec_to_Hex(){
    int input;

    scanf("%d",&input);
    char res[100];
    int j=0;
    while (input!=0) {
        int temp=0;
        temp=input%16;
        if (temp < 10) {
            res[j]=temp+48;
            j++;
        }
        else {
            res[j]=temp+55;
            j++;
        }
        input=input/16;
    }
    printf("Hexa Decimal value : ");
    for (int k=j-1;k>=0;k--)
        printf("%c",res[k]);
}
